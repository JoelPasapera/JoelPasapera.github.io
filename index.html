// Análisis de mercado
void OnTick() {
    double ma = iMA(Symbol(), Period(), 
                        MA_Period, 0, 
                        MODE_SMA, PRICE_CLOSE);
    
    if(Close[0] > ma && !HasPosition()) {
        // Señal de compra detectada
        OpenBuyOrder(LotSize, TakeProfit, StopLoss);
    }
}
